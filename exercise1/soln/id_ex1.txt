db.posts.mapReduce(
	function(){
	var i;
	for(i=0;i<this.tags.length;i++)
	{
		val={count:1, views:this.views};
		emit(this.tags[i],val);
	};
	},
	function(key,values)
	{
		sum={count:0,views:0};
		for(var i=0;i<values.length;i++)
		{
		sum.count=sum.count+values[i].count;
		sum.views=sum.views+values[i].views;
		};
		return sum;
	},
	{
		out:'result'
	}
)
